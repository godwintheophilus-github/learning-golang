````
package main

import (
	"fmt"
)

func main() {

	// single dimensional array
	var x []int = []int{1, 2, 3, 4, 5}
	fmt.Println(x)
	x = append(x, 6)
	fmt.Println(x)

	//multi dimensional array
	var y [][]int = [][]int{{1, 2, 3, 4, 5}, {1, 6, 2, 3, 4, 5, 7}}
	fmt.Println(y)
	y = append(y, []int{1, 6, 2, 3, 4, 5, 7})
	fmt.Println(y)
}
````
